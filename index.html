} function getSheet(name){ const ss = getDb(); let sh = ss.getSheetByName(name); if(!sh){ sh = ss.insertSheet(name); } return sh; } function nowISO(){ return new Date().toISOString(); } function uuid(){ return Utilities.getUuid(); } function parseAmenities(str){ return (str||'').split(/;|,|\n/).map(s=>s.trim()).filter(Boolean); } function parsePhotos(str){ return (str||'').split(/;|,|\n/).map(s=>s.trim()).filter(Boolean); } function corsHeaders(output){ try{ output.setHeader('Access-Control-Allow-Origin','*'); } function findUser(username, password_hash){ const sh = getSheet(SHEETS.USERS); const rows = sh.getDataRange().getValues(); const head = rows.shift(); const idx = Object.fromEntries(head.map((k,i)=>[k,i])); sh.appendRow(row); return id; } /*** HTTP HANDLERS ***/ function doOptions(e){ const out = ContentService.createTextOutput('OK'); return corsHeaders(out); } function doGet(e){ const params = e.parameter || {}; } } function doPost(e){ let payload = {}; try{ payload = JSON.parse(e.postData.contents); }catch(err){} const action = payload.action; let result = { ok:false, action }; try{ if(action === 'addListing'){ const id = insertListing(payload);
